kyrc

A fast, offline, keyboard-only typing test for your terminal. Source & other installers →

Debian / Ubuntu (apt)

curl -fsSL https://abh1nav9.github.io/kyrc/kyrc.gpg.key | sudo gpg --dearmor -o /usr/share/keyrings/kyrc.gpg
echo "deb [signed-by=/usr/share/keyrings/kyrc.gpg] https://abh1nav9.github.io/kyrc/deb stable main" \
  | sudo tee /etc/apt/sources.list.d/kyrc.list
sudo apt update && sudo apt install kyrc

Fedora / RHEL (dnf)

sudo tee /etc/yum.repos.d/kyrc.repo <<'REPO'
[kyrc]
name=kyrc
baseurl=https://abh1nav9.github.io/kyrc/rpm
enabled=1
repo_gpgcheck=1
gpgcheck=0
gpgkey=https://abh1nav9.github.io/kyrc/kyrc.gpg.key
REPO
sudo dnf install kyrc

Other package managers

brew install abh1nav9/tap/kyrc     # macOS / Linux
scoop install kyrc                 # Windows (after: scoop bucket add kyrc https://github.com/abh1nav9/scoop-bucket)
winget install abh1nav9.kyrc       # Windows
yay -S kyrc-bin                    # Arch
snap install kyrc                  # Linux
npm i -g @kyrc/kyrc                # any platform with Node
